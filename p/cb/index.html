class add:
    def __init__(self, initial_value=0):
        # Initialize the internal sum with the provided initial_value.
        self._sum = initial_value

    def __call__(self, value):  
        """
        è®©ç±»å¯ä»¥åƒå‡½æ•°ä¸€æ ·è¢«è°ƒç”¨
        Allows the instance to be called like a function.
        Returns a *new* instance of 'add' with the updated sum,
        enabling method chaining.
        """
        # Create a new instance of 'add' with the current sum plus the new value.
        return add(self._sum + value)

    def __add__(self, other):
        """
        Implements the '+' operator (e.g., addTwo + 5).
        Allows adding a number to the add instance.
        """
        if isinstance(other, (int, float)):
            return self._sum + other
        # If 'other' is another 'add' instance, add their internal sums.
        elif isinstance(other, add):
            return self._sum + other._sum
        else:
            return NotImplemented # Or raise an error for unsupported types

    def __radd__(self, other):
        """
        Implements the right-hand side '+' operator (e.g., 5 + addTwo).
        """
        return self.__add__(other)

    def __int__(self):
        """
        Allows converting the instance to an integer (e.g., int(addTwo)).
        Useful for implicit conversions if needed.
        """
        return self._sum

    def __float__(self):
        """
        Allows converting the instance to a float (e.g., float(addTwo)).
        """
        return float(self._sum)

    def __repr__(self):
        """
        Provides a string representation for debugging.
        When you just type 'addTwo' in the console, this is what gets shown.
        """
        return f"add({self._sum})"

    def __str__(self):
        """
        å­—ç¬¦ä¸²è¡¨è¾¾æ–¹æ³•ï¼Œç›´æ¥print
        Provides a user-friendly string representation.
        """
        return str(self._sum)
